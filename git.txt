//Para clonar el repositorio de github:
git clone ruta_repositorio

//Para saber cuál es el estado del repositorio:
git status

//Para rastrear archivos
git add nombre_archivo

//Para rastrear todos los archivos
git add -A 

//Para guardar los cambios
git commit -m "mensaje descriptivo de los cambios"

//Para mandar los cambios a github
git push

//En resumen, cada que hagamos un cambio importante:
git add -A 
git commit -m "descripción del cambio"
git push


//-----------------R A M A S----------------------

//Para cambiar de rama:
git checkout nombre_rama

//Para listar las ramas:
git branch

//Para crear una nueva rama:
git checkout -b nombre_nueva_rama

//Para actualizar una rama con la versión de github:
git pull

//Para juntar una rama con otra (Desde la rama donde se va a juntar la otra)
git merge nombre_rama_a_juntar

//Para eliminar una rama:
git branch -D nombre_rama

//Ciclo común de trabajo:
//(rama) [comando]
(develop) git pull
(develop) git checkput -b nombre/nueva_rama
(nueva_rama) //Trabajo en la nueva rama
(nueva_rama) git add -A 
(nueva_rama) git commit -m "descripción de los cambios"
(nueva_rama) git push //opcional
(nueva_rama) git checkout develop
(develop) git pull
(develop) git merge nueva_rama
//Si hay conflictos, arreglarlos y hacer commit
